#include "minos.h"
        #include "begos.h"
        #include "roomgames.h"
        
        int main()
        {
            RoomGames roomgames = ROOMGAMES_Start("2023.6.15");
            roomgames.AddMember(minos);
            roomgames.AddMember(begos);
            while (forever)
            {
                roomgames.Print("Hi! We create good games in our room!");
        
                Idea m_idea = MINOS_GET_RANDOM_IDEA();
                Idea b_idea = BEGOS_GET_RANDOM_IDEA();
                Idea good_idea = Idea::Mix(m_idea, b_idea);
        
                Cafe cafe = KAFE_21GRAMS_MAKE_RANDOM_COFFEE();
                Kebab kebab = KEBABCI_MAKE_KUZU_ET();
                Pilav pilav = PILAVCI_MAKE_TAVUK_PILAV();
        
                Program nice_program = MINOS_DO_CODING(cafe, kebab);
                Sketch nice_sketch = BEGOS_DO_DRAWING(cafe, pilav);
        
                Game game = ROOMGAMES_MakeGame(good_idea, nice_program, nice_sketch);
        
                Feedback feedback = ROOMGAMES_PublishGame("web", "itch.io");
        
                Money money = ITCHIO_Pay(feedback);
                roomgames.AddMoney(money);
                roomgames.ShareMoney(minos, begos);
            }
            return 0;
        }
